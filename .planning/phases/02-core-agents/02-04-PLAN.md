      Import: import logging
      logger = logging.getLogger(__name__)

      def resolve(conn, name: str, entity_type: str) -> str | None:
          "Three-tier entity resolution (AD-5). Returns entity id or None."

          # Tier 1: exact match
          row = conn.execute(
              "SELECT id FROM entities WHERE type = ? AND name = ?",
              (entity_type, name)
          ).fetchone()
          if row is not None:
              return row[0]

          # Tier 2: alias match
          row = conn.execute(
              "SELECT e.id FROM entities e "
              "JOIN entity_aliases ea ON ea.entity_id = e.id "
              "WHERE e.type = ? AND ea.alias = ?",
              (entity_type, name)
          ).fetchone()
          if row is not None:
              return row[0]

          # Tier 3: embedding similarity — MVP stub
          logger.debug("resolve: embedding tier not implemented for MVP; returning None for '%s' (%s)", name, entity_type)
          return None
